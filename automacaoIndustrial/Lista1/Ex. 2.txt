bool amarelo = 0;
bool verde = 0;
bool vermelho = 0;

void setup(){  
  	pinMode(5, OUTPUT); 
  	pinMode(4, OUTPUT);
	pinMode(3, OUTPUT);
  	pinMode(1, INPUT);       
}

void loop(){
    verde = 1;
  	digitalWrite(5,1);
	delay(5000);
  	verde = 0;
  	digitalWrite(5,0);
  	amarelo = 1;
  	digitalWrite(4,1);
    delay(3000);
  	amarelo = 0;
  	digitalWrite(4,0);
  	vermelho = 1;
	digitalWrite(3, 1); 
	delay(5000); 
	vermelho = 0;
	digitalWrite(3, 0);
}
